 31class NPTrainer(Executor):
 32    def __init__(
 33        self,
 34        delta=1,
 35        sleep_time=0,
 36        train_task_name=AppConstants.TASK_TRAIN,
 37        submit_model_task_name=AppConstants.TASK_SUBMIT_MODEL,
 38        model_name="best_numpy.npy",
 39        model_dir="model",
 40    ):
 41        # Init functions of components should be very minimal. Init
 42        # is called when json is read. A big init will cause json loading to halt
 43        # for long time.
 44        super().__init__()
 45
 46        if not (isinstance(delta, float) or isinstance(delta, int)):
 47            raise TypeError("delta must be an instance of float or int.")
 48
 49        self._delta = delta
 50        self._model_name = model_name
 51        self._model_dir = model_dir
 52        self._sleep_time = sleep_time
 53        self._train_task_name = train_task_name
 54        self._submit_model_task_name = submit_model_task_name
 55
 56    def handle_event(self, event_type: str, fl_ctx: FLContext):
 57        # if event_type == EventType.START_RUN:
 58        #     Create all major components here. This is a simple app that doesn't need any components.
 59        # elif event_type == EventType.END_RUN:
 60        #     # Clean up resources (closing files, joining threads, removing dirs etc)
 61        pass
 62
 63    def execute(
 64        self,
 65        task_name: str,
 66        shareable: Shareable,
 67        fl_ctx: FLContext,
 68        abort_signal: Signal,
 69    ) -> Shareable:
 70        # Any long tasks should check abort_signal regularly. Otherwise abort client
 71        # will not work.
 72        count, interval = 0, 0.5
 73        while count < self._sleep_time:
 74            if abort_signal.triggered:
 75                return make_reply(ReturnCode.TASK_ABORTED)
 76            time.sleep(interval)
 77            count += interval
 78
 79        if task_name == self._train_task_name:
 80            # First we extract DXO from the shareable.
 81            try:
 82                incoming_dxo = from_shareable(shareable)
 83            except BaseException as e:
 84                self.system_panic(f"Unable to convert shareable to model definition. Exception {e.__str__()}", fl_ctx)
 85                return make_reply(ReturnCode.BAD_TASK_DATA)
 86
 87            # Information about workflow is retrieved from the shareable header.
 88            current_round = shareable.get_header(AppConstants.CURRENT_ROUND, None)
 89            total_rounds = shareable.get_header(AppConstants.NUM_ROUNDS, None)
 90
 91            # Ensure that data is of type weights. Extract model data.
 92            if incoming_dxo.data_kind != DataKind.WEIGHTS:
 93                self.system_panic("Model dex should be of kind DataKind.WEIGHTS.", fl_ctx)
 94                return make_reply(ReturnCode.BAD_TASK_DATA)
 95            np_data = incoming_dxo.data
 96
 97            # Display properties.
 98            self.log_info(fl_ctx, f"Incoming data kind: {incoming_dxo.data_kind}")
 99            self.log_info(fl_ctx, f"Model: \n{np_data}")
100            self.log_info(fl_ctx, f"Current Round: {current_round}")
101            self.log_info(fl_ctx, f"Total Rounds: {total_rounds}")
102            self.log_info(fl_ctx, f"Task name: {task_name}")
103            self.log_info(fl_ctx, f"Client identity: {fl_ctx.get_identity_name()}")
104
105            # Check abort signal
106            if abort_signal.triggered:
107                return make_reply(ReturnCode.TASK_ABORTED)
108
109            # Doing some dummy training.
110            if np_data:
111                if NPConstants.NUMPY_KEY in np_data:
112                    np_data[NPConstants.NUMPY_KEY] += self._delta
113                else:
114                    self.log_error(fl_ctx, "numpy_key not found in model.")
115                    return make_reply(ReturnCode.BAD_TASK_DATA)
116            else:
117                self.log_error(fl_ctx, "No model weights found in shareable.")
118                return make_reply(ReturnCode.BAD_TASK_DATA)
119
120            # We check abort_signal regularly to make sure
121            if abort_signal.triggered:
122                return make_reply(ReturnCode.TASK_ABORTED)
123
124            # Save local numpy model
125            try:
126                self._save_local_model(fl_ctx, np_data)
127            except Exception as e:
128                self.log_error(fl_ctx, f"Exception in saving local model: {e}.")
129
130            self.log_info(
131                fl_ctx,
132                f"Model after training: {np_data}",
133            )
134
135            # Checking abort signal again.
136            if abort_signal.triggered:
137                return make_reply(ReturnCode.TASK_ABORTED)
138
139            # Prepare a DXO for our updated model. Create shareable and return
140            outgoing_dxo = DXO(data_kind=incoming_dxo.data_kind, data=np_data, meta={})
141            return outgoing_dxo.to_shareable()
142        else:
143            # If unknown task name, set RC accordingly.
144            return make_reply(ReturnCode.TASK_UNKNOWN)
145
146    def _load_local_model(self, fl_ctx: FLContext):
147        engine = fl_ctx.get_engine()
148        run_number = fl_ctx.get_prop(FLContextKey.CURRENT_RUN)
149        run_dir = engine.get_workspace().get_run_dir(run_number)
150        model_path = os.path.join(run_dir, self._model_dir)
151
152        model_load_path = os.path.join(model_path, self._model_name)
153        try:
154            np_data = np.load(model_load_path)
155        except Exception as e:
156            self.log_error(fl_ctx, f"Unable to load local model: {e.__str__()}")
157            return None
158
159        model = ModelLearnable()
160        model[NPConstants.NUMPY_KEY] = np_data
161
162        return model
163
164    def _save_local_model(self, fl_ctx: FLContext, model: dict):
165        # Save local model
166        engine = fl_ctx.get_engine()
167        run_number = fl_ctx.get_prop(FLContextKey.CURRENT_RUN)
168        run_dir = engine.get_workspace().get_run_dir(run_number)
169        model_path = os.path.join(run_dir, self._model_dir)
170        if not os.path.exists(model_path):
171            os.makedirs(model_path)
172
173        model_save_path = os.path.join(model_path, self._model_name)
174        if model_save_path:
175            with open(model_save_path, "wb") as f:
176                np.save(f, model[NPConstants.NUMPY_KEY])
177            self.log_info(fl_ctx, f"Saved numpy model to: {model_save_path}")
